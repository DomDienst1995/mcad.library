BufferLoader

An asynchronous buffer loader capable of loading multiple individual sample buffers and buffer lists simultaneously.

new BufferLoader(contextnon-null)

Constructs a buffer loader object.
Parameters:

Example

Load a single sample into a buffer

var url = "../kits/909/kick.wav"

bufferLoader.loadBuffer(url, function(result, url){
				
    // Sample failed to load/decode
    if(result == null) {
        console.log("Single file failed to load " + url);
    }
    // Sample was successfully loaded
    else {
	
        console.log("Finished loading single sample " + url);
    }
});

loadBufferList(bufferList, urlList, onFinishednullable, onLoadnullable)

Loads a sequential array or key/value array of samples into a sequential or key/value array of buffers.
Parameters:
Name Type Attributes Description
bufferList Array.<AudioBuffer> | Array.<key, AudioBuffer> the destination array (sequential or key/value) the sample buffers will be loaded into.
urlList Array.<string> | Array.<key, string> the url array (sequential or key/value) dictating the location of the samples to be loaded.
onFinished function <nullable>
 the event handler to be called when all of the sample buffers requested have been loaded (or attempted).
onLoad function <nullable>
 the event handler to be called when a sample buffer in the array is loaded (or attempted).
Properties
Name Type Description
result object contains the result (key/value pairs) of a given sample buffer's loading/decoding:
Properties
Name Type Description
url string the url of the sample buffer that was loaded (or attempted).
key number | string the index (for sequential) or key (for key/value) of the sample buffer that was loaded (or attempted).
success boolean true if loading/decoding of this buffer was successful, false if not.

Examples

Loading an array of key/value samples

var samples;
 
bufferLoader.loadBufferList(samples, {"kick": "samples/kick.wav", "hat": "samples/hat.wav", "snare": "samples/snare.wav"}, function() {
					
    console.log("Finished loading samples!");
			
    for(key in samples) {

        var buffer = samples[key];
		
        console.log("Sample: " + key + ", buffer: " + buffer);
    }
});

// Elsewhere in the code...
playBuffer(samples.kick); // Or playBuffer(samples["kick"]);

Handling sample load/decode failures

var samples;

bufferLoader.loadBufferList(samples, {"kick": "samples/kick.wav", "hat": "samples/hat.wav", "snare": "samples/snare.wav"}, function() {
    					
    console.log("Finished loading samples!");
			
    for(key in samples) {
        console.log("Sample: " + key + ", buffer: " + samples[key]);
    }
},
function(result) {
				
   if(!result.success) console.log("File " + result.url + " with key " + result.key + " failed to load!");
   else console.log("File " + result.url + " with key " + result.key + " loaded successfully!");
			
});

Using a key/value array specified elsewhere in the code

var samples;

var toLoad = {"kick": "samples/kick.wav", "hat": "samples/hat.wav", "snare": "samples/snare.wav"};

bufferLoader.loadBufferList(samples, toLoad, function() {			
    console.log("Finished loading samples!");
});

Loading an sequential array of samples

var samples;

var urls = ["samples/kick.wav", "samples/hat.wav", "samples/snare.wav"];

bufferLoader.loadBufferList(samples, urls, function() {
					
    for(i = 0; i < samples.length; i++) {
        console.log("Sample: " + i + ", buffer: " + samples[i]);
    }
});

// Elsewhere in the code...
playBuffer(samples[2]); // Plays back snare.wav;

Using event handlers specified elsewhere in the code

var samples;

function finishedEventHandler() {
				
    for(i = 0; i < samples.length; i++) {
        console.log("Sample: " + i + ", buffer: " + samples[i]);
    }
}

function onLoadEventHandler(result) {
				
    if(!result.success) console.log("File " + result.url + " with key " + result.key + " failed to load!");
    else console.log("File " + result.url + " with key " + result.key + " loaded successfully!");
}

// URL arrays can be specified inline
bufferLoader.loadBufferList(samples, ["samples/kick.wav", "samples/hat.wav", "samples/snare.wav"], finishedEventHandler, onLoadEventHandler);

// Elsewhere in the code...
playBuffer(samples[2]); // Plays back snare.wav;
